* Exploring Windowed UPDATE...FROM

#+begin_src sql :eval no
UPDATE "outbound"."LocationJournals" AS TRGTAB
SET
  is_latest = SRCTAB.is_latest
FROM (
SELECT locationjournal_id,
  date = (
    MAX(date) OVER (
      PARTITION BY grts_address, source, activity_group_id, type_subset
  )) AS is_latest
FROM "outbound"."LocationJournals"
) AS SRCTAB
WHERE TRGTAB.locationjournal_id = SRCTAB.locationjournal_id
;
#+end_src

-- Bring it together
SELECT
  LJ.location_id,
  TRUE AS has_loceval,
  LJ.loceval_latest_date,
  LJ.grts_address, -- <- DROP
  LE.grts_address, -- <- DROP
  LE.type AS stratum, -- <- DROP (used to join)
  LJ.loceval_replacement,
  LJ.loceval_type_absence, -- <- DROP (filtered)
  LE.loceval_positive,
  LE.loceval_colleague,
  LE.*, -- <- DROP (filtered)
  LJ.*, -- <- DROP (filtered)
  LE.loceval_notes
FROM (
  SELECT
    location_id,
    grts_address,
    STRING_TO_ARRAY(type_subset, ',') AS types,
    -- (source = 'loceval') AS loceval_latest_by_mnm,
    date AS loceval_latest_date,
    loceval_replacement,
    loceval_type_absence
  FROM "outbound"."LocationJournals"
  WHERE TRUE
    AND category = 'biot'
    AND is_latest
) AS LJ
LEFT JOIN (
  SELECT
    grts_address,
    type,
    eval_date,
    eval_name AS loceval_colleague,
    (  ((type_assessed IS NULL)
       OR (type_assessed = type))
       AND NOT type_is_absent
    ) AS loceval_positive,
    photo AS loceval_photo,
    notes AS loceval_notes
  FROM "outbound"."LocationEvaluations"
  WHERE eval_source = 'loceval'
) AS LE
  ON (LE.grts_address = LJ.grts_address)
  AND (CAST(LE.type AS TEXT) = ANY(LJ.types))
  AND (LJ.loceval_latest_date = LE.eval_date)
WHERE TRUE
  AND NOT loceval_type_absence
  -- AND LE.grts_address IS NULL
  -- AND LJ.grts_address IN (23238, 6314694, 23091910)
  -- AND LJ.grts_address IN (50725598)
ORDER BY location_id ASC
;
